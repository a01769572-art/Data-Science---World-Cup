           p_draw: Series[float] = pa.Field(ge=0, le=1)
           p_away_win: Series[float] = pa.Field(ge=0, le=1)
           outcome_idx: Series[int] = pa.Field(isin=[0, 1, 2])

               return (total - 1.0).abs() <= 1e-6
